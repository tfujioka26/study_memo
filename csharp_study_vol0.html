ディレクトリ内のファイルをリストアップ

1/3 2022

カテゴリー：基本操作

標準モジュールのglobを使います。
検索したい拡張子、フォルダなど指定できます。

サブディレクトリまでリストアップするには、
"**"とrecursive=Trueを記入します。 


                public MainViewModel()

    Items = new ObservableCollection();

                [ObservableProperty]
private string newItemText;

[ObservableProperty]
private ObservableCollection items;

[ObservableProperty]
private string selectedItem;

[RelayCommand]
private void AddItem()
{
    if (!string.IsNullOrWhiteSpace(NewItemText))
    {
        Items.Add(NewItemText);
        NewItemText = string.Empty;
    }
}

[RelayCommand(CanExecute = nameof(CanRemoveItem))]
private void RemoveItem()
{
    if (SelectedItem != null)
    {
        Items.Remove(SelectedItem);
        SelectedItem = null;
    }
}

private bool CanRemoveItem()
{
    return SelectedItem != null;
}

partial void OnSelectedItemChanged(string value) 
{ 
    RemoveItemCommand.NotifyCanExecuteChanged(); 
}

//xaml.cs
private void TextBox_KeyDown(object sender, KeyEventArgs e)
{
    if (e.Key == Key.Enter)
    {
        var viewModel = DataContext as MainViewModel;
        if (viewModel != null && viewModel.AddItemCommand.CanExecute(null))
        {
            viewModel.AddItemCommand.Execute(null);
        }
    }
}
