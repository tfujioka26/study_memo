import numpy as np
# import cv2
# from datetime import datetime


class CreateImage:
def __init__(self, width: int, height: int):
self.height = height
self.width = width

def create_init(self, level: float):
init = np.zeros((self.height, self.width))
init += level
return init

def create_dark_noise(self, sigma: float):
noise = np.random.normal(0, sigma, (self.height, self.width))
return noise

def create_shot_noise(self, level: float):
sigma = np.sqrt(level)
noise = np.random.normal(0, sigma, (self.height, self.width))
return noise

def create_sens_zara(self, level: float, sigma: float, seed_no: int):
sigma = level * sigma / 100.0
np.random.seed(seed_no)
zara = np.random.normal(0, sigma, (self.height, self.width))
return zara

def create_trans_zara(self, sigma: float, seed_no: int):
np.random.seed(seed_no)
zara = np.random.normal(0, sigma, (self.height, self.width))
return zara

def create_image_data(self, level=0.0, dark_noise=0.0,
                sens_zara=0.0, sens_zara_seed_no=1,
                trans_zara=0.0, trans_zara_seed_no=2):
# todo:ショットノイズののせ方、感度ざら付与データに対して実施？
# todo:転送ザラは電荷で変動しない？
image_data = self.create_init(level)
dn = self.create_dark_noise(dark_noise)
sn = self.create_shot_noise(level)
sz = self.create_sens_zara(level, sens_zara, sens_zara_seed_no)
tz = self.create_sens_zara(level, trans_zara, trans_zara_seed_no)
image_data += dn + sn + sz + tz

return image_data

import cv2
from datetime import datetime


def save_image(data, file_name, file_type):
    now = datetime.now().strftime('%y%m%d%H%M%S%f')
    cv2.imwrite(file_name + '_' + now + '.' + file_type, data)
